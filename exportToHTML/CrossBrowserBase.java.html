276      public  void verifyLink(String urlLink) {
277          //Sometimes we may face exception "java.net.MalformedURLException". Keep the code in try catch block to continue the broken link analysis
278          try {
279              //Use URL Class - Create object of the URL Class and pass the urlLink as parameter
280              URL link = new URL(urlLink);
281              // Create a connection using URL object (i.e., link)
282              HttpURLConnection httpConn =(HttpURLConnection)link.openConnection();
283              //Set the timeout for 2 seconds
284              httpConn.setConnectTimeout(2000);
285              //connect using connect method
286              httpConn.connect();
287              //use getResponseCode() to get the response code.
288              if(httpConn.getResponseCode()== 200) {
289                  System.out.println(urlLink+" - "+httpConn.getResponseCode());
290              }
291              if(httpConn.getResponseCode()== 404) {
292                  System.out.println(urlLink+" - "+httpConn.getResponseCode());
293              }
294              if(httpConn.getResponseCode()== 400) {
295                  System.out.println(urlLink+" - "+httpConn.getResponseCode()); }
296              if(httpConn.getResponseCode()== 500) {
297                  System.out.println(urlLink+" - "+httpConn.getResponseCode()); }
298          }
299          //getResponseCode method returns = IOException - if an error occurred connecting to the server.
300          catch (Exception e) {
301          }
302      }
